47  	public void testExecute_fileCannotBeRead() throws Exception {

51  		when(file.exists()).thenReturn(true);
52

54  
55  		assertException(MojoExecutionException.class, "can't be read");

107 	private File getFile(String filename) throws URISyntaxException {
108 		URL resource = this.getClass().getClassLoader().getResource(filename);
109 		return resource != null ? new File(resource.toURI()) : null;
110 	}
111 
112 	private void assertForgotToUpdateTheChangelog() {
113 		assertException(MojoFailureException.class, "forget to update the changelog");
114 	}
115 
116 	private void assertException(Class<? extends AbstractMojoExecutionException> ex, String message) {
117 		assertThatThrownBy(() -> sut.execute()).isInstanceOf(ex).hasMessageContaining(message);
118 	}
